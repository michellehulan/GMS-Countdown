<script>
  function updateCountdown() {
    const now = new Date();
    const y = now.getFullYear();

    // Compute "today" as UTC midnight (prevents timezone/offset issues)
    const todayUTC = Date.UTC(y, now.getMonth(), now.getDate());

    // Jan = 0. Start with this year's Jan 13 at UTC midnight
    let targetYear = y;
    let targetUTC = Date.UTC(targetYear, 0, 13);

    // If Jan 13 this year is before today, use next year
    if (targetUTC < todayUTC) {
      targetYear = y + 1;
      targetUTC = Date.UTC(targetYear, 0, 13);
    }

    const msPerDay = 86400000; // 1000 * 60 * 60 * 24
    const days = Math.ceil((targetUTC - todayUTC) / msPerDay);

    // Render
    document.getElementById('days').textContent =
      `${days} days until Jan 13, ${targetYear}`;
  }

  updateCountdown();
  // Recompute a few times per day (safe for long-lived pages)
  setInterval(updateCountdown, 1000 * 60 * 60 * 6);
</script>
