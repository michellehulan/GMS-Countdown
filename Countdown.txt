<script>
  function updateCountdown() {
    const now = new Date();
    // Target: January 13 of this year
    let target = new Date(now.getFullYear(), 0, 13); // Jan = 0
    target.setHours(0, 0, 0, 0); // normalize to midnight

    // If today is after Jan 13, use next year
    if (now > target) {
      target = new Date(now.getFullYear() + 1, 0, 13);
      target.setHours(0, 0, 0, 0);
    }

    // Normalize 'today' to midnight to avoid partial-day rounding
    const todayMidnight = new Date(now.getFullYear(), now.getMonth(), now.getDate());
    const diffMs = target - todayMidnight;
    const days = Math.ceil(diffMs / (1000 * 60 * 60 * 24));

    document.getElementById('days').textContent =
      `${days} days until Jan 13`;
  }

  updateCountdown();
  setInterval(updateCountdown, 1000 * 60 * 60 * 6); // update every 6 hours
</script>
