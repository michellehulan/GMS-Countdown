<div id="days"></div>

<script>
  function update() {
    const today = new Date();
    let target = new Date(today.getFullYear(), 0, 13); // Jan = 0

    // If Jan 13 has already passed this year, use next year
    if (today > target) {
      target = new Date(today.getFullYear() + 1, 0, 13);
    }

    const diff = target - today;
    const days = Math.ceil(diff / (1000 * 60 * 60 * 24));

    document.getElementById('days').textContent = days + " days until Jan 13";
  }

  update();
  setInterval(update, 1000 * 60 * 60); // update every hour
</script>
